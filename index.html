// 設定基準匯率 (若 Excel 內沒抓到則以此為準) const DEFAULT_RATE_USD = 31.8; function handleFileUpload(input, type) { const file = input.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = function(e) { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, {type: 'array'}); const firstSheet = workbook.Sheets[workbook.SheetNames[0]]; const jsonData = XLSX.utils.sheet_to_json(firstSheet); // 針對「船務」或「貿易」進行特定的加總邏輯 processAccountingData(type, jsonData, file.name); }; reader.readAsArrayBuffer(file); } function processAccountingData(type, data, fileName) { let totalNTD = 0; let totalUSD = 0; let totalJPY = 0; data.forEach(row => { // 依據您的欄位名稱抓取資料 // 使用 Number() 確保字串轉為數字，並處理可能的空值 const usd = Number(row["美金金額"] || 0); const ntd = Number(row["台幣金額"] || 0); const jpy = Number(row["日幣金額"] || 0); // 若未來有日幣欄位 totalUSD += usd; totalNTD += ntd; totalJPY += jpy; }); // 更新介面 const typeKey = type === '貿易' ? 'trade' : (type === '船務' ? 'ship' : 'fish'); document.getElementById(`val-${typeKey}-ntd`).innerText = Math.round(totalNTD).toLocaleString(); document.getElementById(`val-${typeKey}-usd`).innerText = totalUSD.toLocaleString(undefined, {minimumFractionDigits: 2}); document.getElementById(`val-${typeKey}-jpy`).innerText = Math.round(totalJPY).toLocaleString(); document.getElementById(`log-${typeKey}`).innerText = `已解析: ${fileName} (${data.length} 筆)`; // 重新計算總表 calculateGrandTotal(); } function calculateGrandTotal() { const allNtd = parseFormattedNumber('val-trade-ntd') + parseFormattedNumber('val-ship-ntd') + parseFormattedNumber('val-fish-ntd'); const allUsd = parseFormattedNumber('val-trade-usd') + parseFormattedNumber('val-ship-usd') + parseFormattedNumber('val-fish-usd'); const allJpy = parseFormattedNumber('val-trade-jpy') + parseFormattedNumber('val-ship-jpy') + parseFormattedNumber('val-fish-jpy'); // 將所有幣別折算回台幣 (假設匯率抓取自頂部) const grandTotal = allNtd + (allUsd * DEFAULT_RATE_USD) + (allJpy * 0.199); document.getElementById('grand-total').innerText = `NT$ ${Math.round(grandTotal).toLocaleString()}`; // 更新狀態標籤 const status = document.getElementById('status-tag'); status.innerText = "已整併完成"; status.className = "bg-green-500 text-white px-3 py-1 rounded-full text-xs font-bold"; } function parseFormattedNumber(id) { return parseFloat(document.getElementById(id).innerText.replace(/,/g, '')) || 0; }
